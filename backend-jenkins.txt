pipeline {
    agent any

    environment {
        AWS_ACCOUNT_ID = "088042034872"
        AWS_REGION     = "us-east-1"
        ECR_REPO       = "backend"
        IMAGE_NAME     = "backend"
        GIT_REPO_NAME  = "Fullstack-project-nodejs"
        GIT_USER_NAME  = "Gaurav-kangane"
    }

    stages {

        stage('Cleaning Workspace') {
            steps {
                cleanWs()
            }
        }

        stage('Checkout from Git') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/Gaurav-kangane/Fullstack-project-nodejs.git'
            }
        }

        stage("Docker Image Build") {
            steps {
                dir('backend') {
                    sh '''
                        docker system prune -f
                        docker container prune -f
                        docker build -t ${IMAGE_NAME} .
                    '''
                }
            }
        }

        stage("ECR Image Pushing") {
            steps {
                sh '''
                    aws ecr get-login-password --region ${AWS_REGION} \
                    | docker login --username AWS --password-stdin \
                    ${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com

                    docker tag ${IMAGE_NAME}:latest \
                    ${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO}:${BUILD_NUMBER}

                    docker push \
                    ${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO}:${BUILD_NUMBER}
                '''
            }
        }

        stage('Update Deployment file') {
            steps {
                dir('kubernetes-files') {
                    withCredentials([string(credentialsId: 'my-git-pattoken', variable: 'git_token')]) {
                        sh '''
                            git config user.email "gauravkangane2000@gmail.com"
                            git config user.name "Gaurav-kangane"

                            sed -i "s#image:.*#image: ${AWS_ACCOUNT_ID}.dkr.ecr.${AWS_REGION}.amazonaws.com/${ECR_REPO}:${BUILD_NUMBER}#g" backend-deployment.yaml

                            git add .
                            git commit -m "Update backend image to version ${BUILD_NUMBER}"
                            git push https://${git_token}@github.com/${GIT_USER_NAME}/${GIT_REPO_NAME}.git HEAD:main
                        '''
                    }
                }
            }
        }
    }
}
